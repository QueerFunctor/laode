\begin{exercise} \label{c4.3.11}
Let $L$ be a linear map.  Using the definition of linearity,
prove that $L(0)=0$.

\begin{solution}

Let $L(0) = K$.
By the definition of linearity, for any real number $c$,
\[ L(0) = L(c0) = cL(0) \]
which is valid only when $L(0) = 0$.

\end{solution}
\end{exercise}
