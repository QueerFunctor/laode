\begin{exercise} \label{c4.3.2}
Let $x=(4,7)$ and $y=(2,-1)$.  Write the vector $\alpha x+\beta
y$ as a vector in coordinates.

\begin{solution}

$\alpha (4,7) + \beta (2,-1) = (4\alpha + 2\beta, 7\alpha - \beta)$.

\end{solution}
\end{exercise}
