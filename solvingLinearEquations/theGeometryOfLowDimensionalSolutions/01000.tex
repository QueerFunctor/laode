\begin{computerExercise} \label{c2.2.2}
Use \Matlab to solve graphically the planar system of linear
equations
\[
\arraycolsep 2pt
\begin{array}{rcrcr}
4.23x & + & 0.023y & = & -1.1 \\
1.65x & - & 2.81y & = &  1.63
\end{array}
\]
to an accuracy of two decimal points.

\begin{solution}

$(x,y) \approx (-0.26,-0.73)$.


\end{solution}
\end{computerExercise}
