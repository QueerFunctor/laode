\begin{computerExercise} \label{c1.4.5}
Determine the vector of length $1$ that points in the same direction
as the vector
\[
x=(2,13.5,-6.7,5.23).
\]

\begin{solution}

$\frac{x}{||x||} = (0.1244, 0.8397, -0.4167, 0.3253)$.

\end{solution}
\end{computerExercise}
