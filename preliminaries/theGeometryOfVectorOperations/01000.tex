\begin{computerExercise} \label{c1.4.5b}
Determine the vector of length $1$ that points in the same direction
as the vector
\[
y=(2.1,-3.5,1.5,1.3,5.2).
\]

\begin{solution}

\ans $\frac{x}{||x||} = (0.3043, -0.5071, 0.2173, 0.1883, 0.7534)$.

\end{solution}
\end{computerExercise}
