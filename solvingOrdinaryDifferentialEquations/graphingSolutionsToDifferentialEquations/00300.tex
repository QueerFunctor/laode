\noindent In Exercises~\ref{c3.2.3A} -- \ref{c3.2.3D} determine whether the 
given differential equation is autonomous or nonautonomous.
\begin{exercise} \label{c3.2.3A}
$\dot{x}=x-t$.

\begin{solution}
Nonautonomous.

\end{solution}
\end{exercise}
\begin{exercise} \label{c3.2.3B}
$\dot{x}=x^2-x$.

\begin{solution}
Autonomous.

\end{solution}
\end{exercise}
\begin{exercise} \label{c3.2.3C}
$\dot{x}=x\sin(x)$.

\begin{solution}
Autonomous.

\end{solution}
\end{exercise}
\begin{exercise} \label{c3.2.3D}
$\dot{x}=x\cos(t)$.

\begin{solution}
Nonautonomous.



\end{solution}
\end{exercise}
