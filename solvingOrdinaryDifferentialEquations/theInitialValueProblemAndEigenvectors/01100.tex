\begin{computerExercise} \label{c4.5.5b}
$C=\mattwo{1}{1.5}{0}{-2}$.

\begin{solution}
The vector $(-1,2)^t$ is an eigenvector of $C$ with
corresponding eigenvalue $-2$, and the vector $(1,0)^t$ is an eigenvector
with eigenvalue $1$.


\end{solution}
\end{computerExercise}
