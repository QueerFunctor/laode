\begin{exercise} \label{c6.5.5}
Which $n\times n$ matrices are similar to $I_n$?

\begin{solution}

\ans $I_n$ is similar only to itself.

\soln If $A$ is similar to $I_n$, then $A = P^{-1}I_nP = P^{-1}P = I_n$.

\end{solution}
\end{exercise}
